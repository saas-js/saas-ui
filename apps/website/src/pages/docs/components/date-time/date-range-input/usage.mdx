## Import

- `DateRangePicker`: The container component that manages state and context.
- `DateRangePickerInput`: A date range form input.
- `DateRangePickerDialog`: A dialog that contains a calendar and time picker.
- `DateRangePickerCalendar`: A calendar that allows users to select a date range.

#

- `DateRangeInput`: A composite component that composes the above components.

```ts
import {
  DateRangePicker
  DateRangePickerInput,
  DateRangePickerDialog,
  DateRangePickerCalendar,
  DateRangeInput
  } from '@saas-ui/date-picker'
```

## Usage

<Info>
  The DatePicker uses
  [@internationalized/date](https://react-spectrum.adobe.com/internationalized/date/)
  to represent and manipulate dates. [Check out the
  documentation](https://react-spectrum.adobe.com/internationalized/date/) to
  learn how to work with Date objects.
</Info>

### Basic

```jsx center=true
import { FormControl, FormLabel } from '@chakra-ui/react'
import { DateRangeInput } from '@saas-ui/date-picker'

export default function Basic() {
  return (
    <FormControl>
      <FormLabel>Dates</FormLabel>
      <DateRangeInput />
    </FormControl>
  )
}
```

### DateTime

```jsx center=true
import { FormControl, FormLabel } from '@chakra-ui/react'
import { DateRangeInput } from '@saas-ui/date-picker'

export default function DateTime() {
  return (
    <FormControl>
      <FormLabel>Dates</FormLabel>
      <DateRangeInput granularity="minute" />
    </FormControl>
  )
}
```


### Controlled

```jsx
import { FormControl, FormLabel } from '@chakra-ui/react'
import { DateRangeInput } from '@saas-ui/date-picker'

export default function ControlledPicker() {
  const [value, setValue] = React.useState({
    start: today(getLocalTimeZone()),
    end: null,
  })

  return <DateRangeInput value={value} onChange={setValue} />
}
```

### Time zone

```jsx
import { FormControl, FormLabel } from '@chakra-ui/react'
import { DateRangeInput } from '@saas-ui/date-picker'

export default function ControlledPicker() {
  const [value, setValue] = React.useState({
    start: now(getLocalTimeZone()),
    end: null,
  })

  return (
    <DateRangeInput value={value} onChange={setValue} granularity="minute" />
  )
}
```

### Hide time zone

```jsx
import { FormControl, FormLabel } from '@chakra-ui/react'
import { DateRangeInput } from '@saas-ui/date-picker'

export default function ControlledPicker() {
  const [value, setValue] = React.useState({
    start: now(getLocalTimeZone()),
    end: null,
  })

  return (
    <DateRangeInput
      value={value}
      onChange={setValue}
      granularity="minute"
      hideTimeZone
    />
  )
}
```

### Composed

```jsx
import { FormControl, FormLabel } from '@chakra-ui/react'
import {
  DateRangePicker,
  DateRangeInput,
  DateRangePickerDialog,
  DateRangePickerCalendar,
} from '@saas-ui/date-picker'

export default function Composed() {
  return (
    <FormControl>
      <FormLabel>Dates</FormLabel>
      <DateRangePicker>
        <DateRangeInput />
        <DateRangePickerDialog>
          <DateRangePickerCalendar />
        </DateRangePickerDialog>
      </DateRangePicker>
    </FormControl>
  )
}
```
