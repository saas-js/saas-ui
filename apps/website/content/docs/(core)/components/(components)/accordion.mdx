## Anatomy

```js title="Parts"
import { Accordion } from '@saas-ui/react/accordion'

function Parts() {
  return (
    <Accordion.Root>
      <Accordion.Item>
        <Accordion.ItemTrigger />
        <Accordion.ItemContent />
      </Accordion.Item>
    </Accordion.Root>
  )
}
```
