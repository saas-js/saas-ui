## Usage

```jsx
import { GridList } from '@saas-ui/react/grid-list'
```

```jsx
<GridList.Root>
  <GridList.Header>Header</GridList.Header>
  <GridList.Item>
    <GridList.Cell>Cell</GridList.Cell>
  </GridList.Item>
</GridList.Root>
```
